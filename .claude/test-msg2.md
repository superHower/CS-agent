 function isSystemMessage(message, kefu) {
    if (typeof message !== 'string' || !message.trim()) {
        return true;
    }

    // 静态通用系统关键词
    const staticKeywords = [
        '系统消息', '系统自动发送', '机器人发送', '机器人接待中',
        '用户超时未回复，系统关闭会话', '平台已自动同意', '售后小助手',
        '系统自动同意', '消费者正在查看订单', '平台主动处理',
        '邀请下单', '商家配置发送', '系统关闭会话',
        '当前会话已长时间未回复', '退款成功', '同意退款',
        '支付提醒', '订单已关闭', '消费者催发货',
        '从历史会话发起会话', '平台已自动同意补寄',
        '用户仍在等待您的处理结果'
    ];

    // 动态拼接客服专属系统关键词
    const dynamicKeywords = [];
    if (typeof kefu === 'string' && kefu.trim()) {
        dynamicKeywords.push(`客服${kefu}接入`);
        dynamicKeywords.push(`${kefu}撤回了一条消息`);
        dynamicKeywords.push(`${kefu}撤回了一条消息，已被编辑`);
    }

    // 合并后统一做全文包含匹配
    const allKeywords = [...staticKeywords, ...dynamicKeywords];
    return allKeywords.some(kw => message.includes(kw));
}

function convertChatToMessages(chatList, kefu) {
    return chatList.filter(item => !isSystemMessage(item, kefu));
}
